201 template<class Type>
202 void Foam::spaceTimeWindowFvPatchField<Type>::checkTable()
203 {
204  if (sampleTimes_.empty())
205  {
206  findSampleTimes();
207  startSampleTime_ = -1;
208  endSampleTime_ = -1;
209  }
210 
211  const scalar currentTime = this->db().time().value();
212 
213  // Find bracketing times
214  label lo = -1;
215  label hi = -1;
216 
217  for (label i = 0; i < sampleTimes_.size(); ++i)
218  {
219  if (sampleTimes_[i].value() <= currentTime)
220  {
221  lo = i;
222  }
223  if (sampleTimes_[i].value() >= currentTime && hi < 0)
224  {
225  hi = i;
226  }
227  }
228 
229  // Strict time bounds - NO extrapolation allowed
230  if (lo < 0)
231  {
232  FatalErrorInFunction
233  << "Current time " << currentTime
234  << " is before first available sample time "
235  << sampleTimes_[0].value() << nl
236  << " Patch: " << this->patch().name() << nl
237  << " This BC requires boundary data to be pre-computed" << nl
238  << " (run spaceTimeWindowExtract first)" << nl
239  << exit(FatalError);
240  }
241  if (hi < 0)
242  {
243  FatalErrorInFunction
244  << "Current time " << currentTime
245  << " is after last available sample time "
246  << sampleTimes_.last().value() << nl
247  << " Patch: " << this->patch().name() << nl
248  << " This BC requires boundary data to be pre-computed" << nl
249  << " (ensure extraction covers full simulation time)" << nl
250  << exit(FatalError);
251  }
252 
253  // Check if we need to load new data
254  if (lo != startSampleTime_)
255  {
256  startSampleTime_ = lo;
257  startSampledValues_ = readFieldData(sampleTimes_[lo]);
258 
259  DebugInfo
260  << "Loaded start time data: " << sampleTimes_[lo].name() << endl;
261  }
262 
263  if (hi != endSampleTime_)
264  {
265  endSampleTime_ = hi;
266  endSampledValues_ = readFieldData(sampleTimes_[hi]);
267 
268  DebugInfo
269  << "Loaded end time data: " << sampleTimes_[hi].name() << endl;
270  }
271 }

573 template<class Type>
574 void Foam::spaceTimeWindowFvPatchField<Type>::updateCoeffs()
575 {
576  if (this->updated())
577  {
578  return;
579  }
580 
581  // Validate extraction metadata (once)
582  validateMetadata();
583 
584  // Load/check sample times and data
585  checkTable();
586 
587  const scalar currentTime = this->db().time().value();
588 
589  // Compute interpolated values
590  Field<Type>& patchValues = *this;
591 
592  if (startSampleTime_ == endSampleTime_)
593  {
594  // No interpolation needed - exact time match or at boundary
595  patchValues = startSampledValues_ + offset_;
596  }
597  else
598  {
599  // Linear temporal interpolation
600  const scalar t0 = sampleTimes_[startSampleTime_].value();
601  const scalar t1 = sampleTimes_[endSampleTime_].value();
602  const scalar alpha = (currentTime - t0) / (t1 - t0 + SMALL);
603 
604  patchValues =
605  (1.0 - alpha) * startSampledValues_
606  + alpha * endSampledValues_
607  + offset_;
608 
609  DebugInfo
610  << "Time interpolation: t=" << currentTime
611  << " between " << t0 << " and " << t1
612  << " alpha=" << alpha << endl;
613  }
614 
615  fixedValueFvPatchField<Type>::updateCoeffs();
616 }
