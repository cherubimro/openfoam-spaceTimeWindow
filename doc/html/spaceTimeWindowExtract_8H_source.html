14  Foam::functionObjects::spaceTimeWindowExtract

16 Group
17  grpFieldFunctionObjects

19 Description
20  Function object to extract space-time window data during simulation.
21 
22  Outputs:
23  - Subset mesh (once, at first write)
24  - Face-interpolated boundary values at each timestep
25 
26  The face values are computed using linear interpolation between
27  adjacent cell centers, which is exact for OpenFOAM's cell-centered FVM.
28 
29  Output structure:
30  \verbatim
31  outputDir/
32  constant/
33  polyMesh/ # Subset mesh files
34  boundaryData/
35  oldInternalFaces/
36  points # Face centres (for reference only)
37  <time>/
38  U # Face-interpolated values
39  p
40  <startTime>/ # Initial subset fields
41  U
42  p
43  \endverbatim
44 
45 Usage
46  \verbatim
47  spaceTimeWindowExtract1
48  {
49  type spaceTimeWindowExtract;
50  libs (spaceTimeWindow);
51 
52  // Define region directly with a box (fully internal to domain)
53  box (0.05 -0.25 0.01) (0.90 0.25 0.38);
54 
55  outputDir "../ufr2-02-subset"; // Output case directory
56  fields (U p); // Fields to extract
57 
58  // Optional: write format for boundary data
59  // Options: ascii, binary, deltaVarint
60  // Default: ascii
61  // deltaVarint provides ~2.7% compression ratio (vs raw binary)
62  writeFormat deltaVarint;
63 
64  // Optional: precision for deltaVarint (significant digits)
65  // Default: 6 (matches OpenFOAM's default writePrecision)
66  deltaVarintPrecision 6;
67 
68  // Optional: compression (gzip) - ignored for deltaVarint
69  writeCompression on;
70 
71  // Optional: encryption with libsodium (requires FOAM_USE_SODIUM=1 build)
72  // encrypt true;
73  // publicKey "base64-encoded-32-byte-public-key";
74 
75  writeControl timeStep;
76  writeInterval 1;
77  }
78  \endverbatim
